# 选择特征和目标变量
X = df[['IQIYI']]
y = df['BOX']

# 创建线性回归模型
model = LinearRegression()
model.fit(X, y)
y_pred = model.predict(X)

# 输出回归的基本信息
print('线性回归基本信息:')
print('斜率 (Coefficient):', model.coef_[0])
print('截距 (Intercept):', model.intercept_)
print('均方误差 (Mean Squared Error):', metrics.mean_squared_error(y, y_pred))
print('R平方值 (R-squared):', metrics.r2_score(y, y_pred))

# 绘制散点图和回归线
plt.figure(figsize=(12, 6),dpi=150)
plt.scatter(X, y, color='gray')
plt.plot(X, y_pred, color='red', linewidth=2)
plt.xlabel('IQIYI Score')
plt.ylabel('Box Office')
plt.title('IQIYI Score vs. Box Office - Linear Regression')
plt.show()
